def makeCorrelationplot(xcol,ycol,df,toPaint):
    df.plot(x=xcol,y=ycol,kind="scatter",color='lightcoral', marker='D',ax=toPaint)
    xvec = df[xcol]
    yvec = df[ycol]
    m, b = np.polyfit(xvec, yvec, 1)
    toPaint.plot(xvec,m*xvec+b,ls='-')
